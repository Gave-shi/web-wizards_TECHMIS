CREATE DATABASE UOR;
USE UOR;
CREATE TABLE STUDENT(
	STUD_ID VARCHAR(5) NOT NULL,
	Level VARCHAR(4),
	Address VARCHAR(50),
	USER_ID VARCHAR(6),
	COURSE_ID VARCHAR(5),
	PRIMARY KEY (STUD_ID),
	FOREIGN KEY (USER_ID) REFERENCE USER(USER_ID),
	FOREIGN KEY (COURSE_ID) REFERENCE COURSE(COURSE_ID),
	);
	
CREATE TABLE USER (
	USER_ID VARCHAR(6) NOT NULL,
	FIRST NAME VARCHAR(25),
	LAST NAME VARCHAR(25),
	PASSWORD VARCHAR(10),
	EMAIL VARCHAR(30),
	CONTACT INT,
	PRIMARY KEY (USER_ID)
);

CREATE TABLE MEDICAL (
	DOC_REG_NO VARCHAR(10)NOT NULL,
	REF_NO VARCHAR(10) NOT NULL,
	DATE DATE,
	PRIMARY KEY (DOC_REG_NO,REF_NO),
	FOREIGN KEY (STUD_ID) REFERENCE STUDENT(STUD_ID)
);

CREATE TABLE EXAM-RESULTS(
	MARKS_ID VARCHAR(5),
	GRADE VARCHAR(2),
	QUIZ1 INT,
	QUIZ2 INT,
	QUIZ3 INT,
	ASSIGMENT MARKS INT,
	THEORY INT,
	PRACTICAL INT,
	PRIMARY KEY (MARKS_ID)
);


CREATE TABLE LECTURE (
	REG_NO VARCHAR(5) NOT NULL,
	LEC_ID VARCHAR(5),
	USER_ID VARCHAR(6),
	Address VARCHAR(50),
	PRIMARY KEY (REG_NO,LEC_ID ),
	FOREIGN KEY (USER_ID) REFERENCE USER(USER_ID)
);
